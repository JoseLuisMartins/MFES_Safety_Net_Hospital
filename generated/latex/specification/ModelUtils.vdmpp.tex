(*@
\label{dateToNat:48}
@*)
 static public dateToNat: Date -> nat
 dateToNat(d) == (
  d.year * 100000000 + 
  d.month * 1000000 + 
  d.day * 10000 + 
  d.hour * 100 + 
  d.min 
 );
